    # Format Number

    function formatNumber(string, delimiter) {
	var delimiter = delimiter ? delimiter : '.';
	var dot = delimiter == '.' ? ',' : '.';
	var ignoredChars = new RegExp(`[^0-9${'\\'+dot}]`, 'g');
	var splitted = String(string).replace(ignoredChars, '').split(new RegExp(`${'\\'+dot}(.+)`));
	var decimal = splitted[0].replace(/(\d)(?=(\d{3})+$)/g, '$1' + delimiter);
	var float = '';

	if(splitted[1]) {
		float = dot + splitted[1].replace(/\D/g, '');
	}

	return ( String(decimal) + String(float) );
};

// Usage
formatNumber('10000.5', ',') // 1,000.5
formatNumber('1.500.000', '.') // 1.500.000
